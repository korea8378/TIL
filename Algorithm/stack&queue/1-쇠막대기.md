import java.util.Stack;

class Solution {
    public int solution(String arrangement) {

        Stack<Integer> stack = new Stack<>();

        int size = arrangement.length();

        int cnt = 0;

        for(int i = 0; i < size; i++) {
            if(arrangement.charAt(i) == '(') {
                stack.push(i);
            } else if(arrangement.charAt(i) == ')') {
                if((stack.pop() + 1) == i) {
                    cnt += stack.size();
                } else {
                    cnt++;
                }
            }
        }

        return cnt;
    }
}
