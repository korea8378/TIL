#### 반복계
* 단점
  * SQL 실행의 오버 헤드 : 전처리, 후처리

  * 예상 처리 시간의 정밀도
  * 트랜잭션 제어가 편리

#### 반복계를 포장계로 변환

INSERT INTO Sales2

							OVER ( PARTITION BY company

#### 우편번호 순위 붙이기 문제(집합계에서 반복계 처리)
<pre><code>
SELECT pcode,
	      distict_name,
               CASE WHEN pcode = ’4130033’ THEN 0
			 WHEN pcode LIKE ‘413003%’ THEN 1
			 WHEN pcode LIKE ‘41300%’ THEN 2
			 WHEN pcode LIKE ‘4130%’ THEN 3
